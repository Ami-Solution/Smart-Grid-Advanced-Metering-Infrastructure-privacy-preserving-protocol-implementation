178 //! Concatonate two strings (stored as CryptoPP::Integer) and hash them.
179 static Integer concatAndHash(HashTransformation* hash, Integer one, Integer two)
180 {
181  byte* input = new byte[one.ByteCount() + two.ByteCount()];
182  for (unsigned int i=0;i<(one.ByteCount() + two.ByteCount());i++) input[i] = 0;
183 
184  for (unsigned int i=0;i<one.ByteCount();i++) input[i] = one.GetByte(i);
185  for (unsigned int i=0;i<two.ByteCount();i++) input[i+one.ByteCount()] = two.GetByte(i);
186 
187  return generateHash(hash, input, one.ByteCount()+two.ByteCount());
