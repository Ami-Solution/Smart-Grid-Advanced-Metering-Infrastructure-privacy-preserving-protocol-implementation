130 // As a one-time operation, UC has to establish the necessary cryptographic
131 // primitives and its private/public keys. UC generates two primes p and
132 // q such that q|p − 1. UC picks a generator g of Z ∗ p of order q. UC
133 // picks uniformly at random x ∈ Z ∗ q as its private key and computes y =
134 // g x mod p as its public key. UC selects hash functions H 1 : {0, 1} ∗ × Z ∗ p →
135 // Z ∗ q , H 2 : {0, 1} l 0 × {0, 1} l 1 → Z ∗ q and H 3 : Z ∗ p × Z ∗ p → {0, 1} l , where
136 // l = l 0 + l 1 ∈ N. UC keeps secret its master key, where masterKey =
137 // (p, q, g, x, H 1 , H 2 , H 3 ).
138 
139 // 1) Generate two primes p and q such that (p - 1) is divisable by q. For example, p = 7 and q = 3, (p - 1) == 0 (mod q).
140 // 2) Pick generator g
141 // 3) Generate x as an integer of order q and compute y.
142 // 4) Shoose hashes, H1 and H2 == SHA-256, H3 == SHA-1 (bitwise, string/hex)
143 // 4.1) l == 160bits,
144 // 4.2) masterKey stores just p,q,g,x
145 // 5) Given ID, generate partial keys...
146 
147 // 2^i for i={1,2,3,4,5}
148 // 2^i mod 9
149 
150 // 2,4,5,7,8

152 bool UtilityCompany::generateKey(Integer q,
153  const Integer & max,
154  const Integer & equiv,
155  const Integer & mod)
156 {
157  pSelector* ps = new pSelector();
158 
159  //http://www.cryptopp.com/docs/ref/nbtheory_8h.html
160  //p is the starting value
161  //max is the maximum value
162  //equiv is the value which the prime will have once it is mod'ed with the mod parameter
163 
164  //this function returns true if x % mod == equiv where x is prime
165  //equiv cannot be greater than or equal to mod
166 
167  //start = std::chrono::high_resolution_clock::now();
168  bool first = FirstPrime(q,max,equiv,mod,ps);
169 // std::cout << "Time to find first prime (q): " << std::chrono::duration_cast<std::chrono::duration<long double>>(std::chrono::high_resolution_clock::now() - start).count() << std::endl;
170  if (!first) return false;
171 
172  Integer p = Integer::Power2(2047)+1;
173 // start = std::chrono::high_resolution_clock::now();
174  bool second = FirstPrime(p,Integer::Power2(2048),1,q,ps);
175  // std::cout << "Time to find second prime (p): " << std::chrono::duration_cast<std::chrono::duration<long double>>(std::chrono::high_resolution_clock::now() - start).count() << std::endl;
176  if (!second) return false;
177 
178  //This may be done by setting g = h^((p–1)/q) mod p for some arbitrary h (1 < h < p−1), and trying again with a different h if the result comes out as 1. Most choices of h will lead to a usable g; commonly h=2 is used.
179  Integer g = 0;
180  Integer h = 2;
181  do {g = ModularExponentiation(h++,((p-1)/q),p) ;} while (g == 1 && h < p-1) ;
182  if (h == p-1) return false;
183  //we have our primes, time to generate keys
184  Integer x = Integer(*rng,1,q);
185  //Public Key
186  Integer y = ModularExponentiation(g, x, p);
187  //std::cout << "uc_init " << std::chrono::duration_cast<std::chrono::duration<long double>>(std::chrono::high_resolution_clock::now() - start).count() << std::endl;
188  keys.priv.partial = x;
189  keys.pub.partial = y;
190  setMasterKey(p,q,g,x);
191  return true;
