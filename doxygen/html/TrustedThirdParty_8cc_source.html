39 void TrustedThirdParty::timedHandleMessage(cMessage *msg)
40 {
41  // Check if the message is for registration at UC
42  if (!strcmp(msg->getName(), "register"))
43  {
44  registerAtUC(msg);
45 
46  return;
47  }
48  // Receiving registration information from the utility company
49  else if (!strcmp(msg->getName(), "registerInfoFromUC"))
50  {
51  registerInfoFromUC(msg);
52 
53  return;
54  }
55  //starting session key exchange
56  else if (!strcmp(msg->getName(), "startSessionKeyExchange"))
57  {
58  startSessionKeyExchange();
59 
60  return;
61  }
62  // Receiving request for a session key exchange phase
63  else if (!strcmp(msg->getName(), "sessionKeyExchange"))
64  {
65  finishSessionKeyExchange(msg);
66 
67  return;
68  }
69  else if (!strcmp(msg->getName(), "registerSM"))
70  {
71  registerSM(msg);
72 
73  return;
74  }
75  processDataFromSM(msg);
76 }
